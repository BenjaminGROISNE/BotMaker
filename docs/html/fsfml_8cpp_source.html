52void fsfml::eventLoop()
53{
54 sf::Event event;
55 while (window->pollEvent(event))
56 {
57 switch (event.type)
58 {
59 case sf::Event::Closed:
60 closeWindow = true;
61 break;
62 case sf::Event::MouseButtonReleased:
63 switch (event.mouseButton.button) {
64 case sf::Mouse::Left:
65 clickPos = mousePos;
66 showVector(convertf(clickPos));
67 hasPressedClick = false;
68 hasClicked = true;
69 break;
70 }
71 break;
72 case sf::Event::MouseButtonPressed:
73 switch (event.mouseButton.button) {
74 case sf::Mouse::Left:
75 clickPos = mousePos;
76 showVector(convertf(clickPos));
77 hasPressedClick = true;
78 break;
79 }
80 break;
81 case sf::Event::KeyReleased:
82 switch (event.key.code) {
83 case sf::Keyboard::A:
84 break;
85 }
86 break;
87 case sf::Event::KeyPressed:
88 break;
89 case sf::Event::MouseMoved:
90 mouseMoved = true;
91 break;
92 case sf::Event::MouseEntered:
93 mouseInWindow = true;
94 break;
95 case sf::Event::MouseLeft:
96 mouseInWindow = false;
97 break;
98 case sf::Event::TextEntered:
99 if (event.text.unicode < 128) {
100 charEntered = static_cast<char>(event.text.unicode);
101 charTyped = true;
102 std::cout << "SFML PART" << charEntered << "\n";
103 }
104 break;
105 case sf::Event::Resized:
106 size = convertf(window->getSize());
107 view.setCenter(size / 2.f);
108 view.setSize(size);
109 window->setSize(convertu(size));
110 window->setView(view);
111 windowResized = true;
112 break;
113 default:
114 break;
115 }
116 }

455bool fsfml::OnSprite(sf::Sprite& sprite, sf::Vector2f Direction)
456{
457 sf::Vector2f topCorner(sprite.getPosition());
458 sf::Vector2f bottomCorner(topCorner.x + sprite.getGlobalBounds().width, topCorner.y + sprite.getGlobalBounds().height);
459 bool belowtopcorner = topCorner.x <= Direction.x && topCorner.y <= Direction.y;
460 bool abovebottomcorner = Direction.x <= bottomCorner.x && Direction.y <= bottomCorner.y;
461 return (belowtopcorner && abovebottomcorner);
462}
