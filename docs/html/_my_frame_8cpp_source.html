502void MyFrame::CalcCreateInsertInFlowPanel(wxPoint mouseClick,FlowTagPanel* flowPanel) {
503 wxSize tagLineSize = flowPanel->getLineSizer()->GetSize();
504 wxPoint tagPanelPos = flowPanel->GetPosition();
505 wxSize panelSize = flowPanel->GetSize();
506 int lineHeight = tagLineSize.y;
507 int panelHeight = panelSize.y;
508 int mouseHeight = mouseClick.y;
509 int half = lineHeight / 2;
510
511 if (mouseHeight < half) {
512 CreateInsertPanel(tagToAdd, flowPanel, ABOVE);
513 }
514 else if (mouseHeight > half && mouseHeight < lineHeight) {
515 std::vector<TagPanel*> listChildren = flowPanel->getChildrenTags();
516 if (!listChildren.empty()) {
517 CreateInsertPanel(tagToAdd, listChildren.front(), ABOVE);
518 }
519 }
520 else if (mouseHeight > panelHeight - lineHeight && mouseHeight < panelHeight - half) {
521 std::vector<TagPanel*> listChildren = flowPanel->getChildrenTags();
522 if (!listChildren.empty()) {
523 CreateInsertPanel(tagToAdd, listChildren.back(), UNDER);
524 }
525 }
526 else if (mouseHeight > panelHeight - half && mouseHeight < panelHeight) {
527 CreateInsertPanel(tagToAdd, flowPanel, UNDER);
528 }
529 resetInsertHover();

612 break;
613 case BOOLTAG:
614 newPanel = new BoolTagPanel(parentTag);
615 break;
616 default:
617 newPanel = new TagPanel(parentTag);
618 break;
619 }
620 if (newFlowPanel) {
621 BindFlowTagPanelEvents(newFlowPanel);
622 newPanel = newFlowPanel;
623 }
624 else if (newPanel) {
625 BindTagPanelEvents(newPanel);

627 newPanel->setTag(tag);
628 return newPanel;

719 else if (mouseHeight > half && mouseHeight < lineHeight) {
720 h = MIDTOPHOVER;

723 h = MIDBOTHOVER;

731 wxSize panelSize = panel->GetSize();
732 int panelHeight = panelSize.GetHeight();
733 if (mouseHeight < (panelHeight / 2)+1) {
734 h = TOPHOVER;
735 }
736 else if (mouseHeight > panelHeight / 2 && mouseHeight < panelHeight) {
737 h = BOTTOMHOVER;
738 }
739 }

741 if (HoverSwitch) {
742 if (hoverNeighbour) {
743 hoverNeighbour->isInserting = false;
744 hoverNeighbour->Refresh();
745 }
746 panel->isInserting = true;
747 if(hoverNeighbour!=panel)panel->Refresh();
748
749 setNeighboursPanel(panel, h);
750 if(hoverNeighbour)hoverNeighbour->Refresh();
751 HoverSwitch = false;
752 }
753 }

933 int newHeight = imageHeight*ratio;
