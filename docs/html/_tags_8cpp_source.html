705void ListTag<T>::execute() {
706 for (auto& tag : listTags) {
707 if (tag) {
708 switch (this->myType) {
709 case LOADTAG: {
710 std::shared_ptr<LoadTag<std::vector<T>>> ldTagList = getLoadTag<std::vector<T>>(tag);
711 std::shared_ptr < LoadTag<T>> ldTag = getLoadTag<T>(tag);
712 if (ldTagList) {
713 ldTagList->execute();
714 myList.insert(myList.end(), ldTagList->value.begin(), ldTagList->value.end());
715 }
716 else if (ldTag) {
717 ldTag->execute();
718 myList.push_back(ldTag->value);
719 }
720 break;
721 }
722 case LISTTAG: {
723 std::shared_ptr <ListTag<T>> liTag = getListTag<T>(tag);
724 if (liTag) {
725 liTag->execute();
726 myList.insert(myList.end(), liTag->myList.begin(), liTag->myList.end());
727 }
728 break;
729 }
730 default: {
731 executeSimpleTag(tag);
732 break;
733 }
734 }
735 }
