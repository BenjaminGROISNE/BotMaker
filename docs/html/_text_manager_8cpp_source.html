156 int nbOpen = 0;
157 int nbClose = 0;
158 std::string copy(text);
159 std::string save;
160 size_t lastClosePos = std::string::npos;
161 size_t firstOpenPos = getSequencePos(copy, open);
162 if (firstOpenPos != std::string::npos) {
163 while (true) {
164 lastClosePos = getSequencePos(copy, close);
165 if (lastClosePos != std::string::npos) {
